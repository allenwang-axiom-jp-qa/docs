```javascript JavaScript
const response = await fetch('https://dev-admin.beauty-666.com/api/v1/oms/diamonds/{YOUR_ID}', {
```

```python Python
import requests

url = 'https://dev-admin.beauty-666.com/api/v1/oms/diamonds/{YOUR_ID}'

    'Authorization': 'Bearer YOUR_TOKEN',
```
