---
title: '获取单个币种'
api: 'GET https://dev-admin.beauty-666.com/api/v1/oms/exchange-rates/{id}'
description: '根据 ID 获取单个币种详情'
---

根据币种 ID 获取单个币种的详细信息。

## 请求头

<ParamField header="Authorization" type="string" required>
  JWT Bearer token，格式：`Bearer YOUR_TOKEN`
</ParamField>

## 路径参数

<ParamField path="id" type="integer" required>
  币种 ID
</ParamField>

## 响应

<ResponseField name="code" type="integer">
  响应状态码，0 表示成功
</ResponseField>

<ResponseField name="data" type="object">
  币种详情

  <Expandable title="币种对象">
    <ResponseField name="id" type="integer">
      币种 ID
    </ResponseField>
    <ResponseField name="code" type="string">
      币种代码
    </ResponseField>
    <ResponseField name="name" type="string">
      币种名称
    </ResponseField>
    <ResponseField name="symbol" type="string">
      货币符号
    </ResponseField>
    <ResponseField name="currency_type" type="integer">
      币种类型：0=法币，1=加密货币
    </ResponseField>
    <ResponseField name="exchange_rate" type="string">
      对 CNY 的汇率
    </ResponseField>
    <ResponseField name="update_mode" type="integer">
      更新模式：0=手动，1=自动同步
    </ResponseField>
    <ResponseField name="last_updated_at" type="string">
      最后更新时间
    </ResponseField>
    <ResponseField name="status" type="integer">
      状态：0=停用，1=启用
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="msg" type="string">
  响应消息
</ResponseField>

<RequestExample>

```bash cURL
curl "https://dev-admin.beauty-666.com/api/v1/oms/exchange-rates/10" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const currencyId = 10;
const response = await fetch(
  `https://dev-admin.beauty-666.com/api/v1/oms/exchange-rates/${currencyId}`,
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN'
    }
  }
);
const data = await response.json();
```

```python Python
import requests

currency_id = 10
response = requests.get(
    f'https://dev-admin.beauty-666.com/api/v1/oms/exchange-rates/{currency_id}',
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)
data = response.json()
```

</RequestExample>
