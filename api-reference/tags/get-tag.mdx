---
title: '获取单个标签'
api: 'GET https://dev-admin.beauty-666.com/api/v1/site-svc-video-mgmt/tag'
description: '根据标签ID获取单个标签详情'
---

根据标签ID获取单个标签的完整信息。

## 查询参数

<ParamField query="tagId" type="integer" required>
  标签数据库主键ID
</ParamField>

## 响应

<ResponseField name="code" type="integer">
  业务状态码（0表示成功）
</ResponseField>

<ResponseField name="msg" type="string">
  响应消息
</ResponseField>

<ResponseField name="data" type="object">
  标签详情
</ResponseField>

<ResponseField name="data.id" type="integer">
  数据库主键ID
</ResponseField>

<ResponseField name="data.tagId" type="string">
  标签唯一标识符，格式：TAG001
</ResponseField>

<ResponseField name="data.name" type="string">
  标签名称
</ResponseField>

<ResponseField name="data.categoryId" type="integer">
  所属分类ID
</ResponseField>

<ResponseField name="data.categoryName" type="string">
  所属分类名称
</ResponseField>

<ResponseField name="data.tagType" type="integer">
  标签类型：0=普通类型，1=热门类型
</ResponseField>

<ResponseField name="data.useCount" type="integer">
  使用次数
</ResponseField>

<ResponseField name="data.videoCount" type="integer">
  关联视频数量
</ResponseField>

<ResponseField name="data.status" type="integer">
  状态：0=禁用，1=启用
</ResponseField>

<ResponseField name="data.sortOrder" type="integer">
  排序顺序
</ResponseField>

<ResponseField name="data.description" type="string">
  标签描述
</ResponseField>

<ResponseField name="data.createdTime" type="integer">
  创建时间（Unix时间戳，秒）
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://dev-admin.beauty-666.com/api/v1/site-svc-video-mgmt/tag?tagId=1" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://dev-admin.beauty-666.com/api/v1/site-svc-video-mgmt/tag?tagId=1', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_TOKEN',
    'Content-Type': 'application/json'
  }
});

const data = await response.json();
```

```python Python
import requests

url = 'https://dev-admin.beauty-666.com/api/v1/site-svc-video-mgmt/tag'
params = {'tagId': 1}
headers = {
    'Authorization': 'Bearer YOUR_TOKEN',
    'Content-Type': 'application/json'
}

response = requests.get(url, params=params, headers=headers)
result = response.json()
```
