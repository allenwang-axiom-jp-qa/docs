<ParamField query="status" type="integer">
  状态筛选
</ParamField>

<ParamField query="category_id" type="integer">
  分类ID筛选
</ParamField>

<ParamField query="tag_ids" type="string">
  标签ID筛选（逗号分隔）
</ParamField>

<ParamField query="search" type="string">

<ParamField query="page" type="integer" default="1">

</ParamField>
