更新模块的标题、配置、排序和启用状态

## 路径参数

<ParamField path="moduleId" type="integer" required>
  模块ID
</ParamField>


## 请求体

<ParamField body="title" type="string">
  模块标题，1-100字符，示例: `新的标题`
</ParamField>

<ParamField body="config" type="string">
  模块配置JSON字符串，示例: `{"autoplay":false,"interval":5000,"minCount":2,"maxCount":6}`
</ParamField>

<ParamField body="order" type="integer">
  显示顺序，0-9999，示例: `2`
</ParamField>

<ParamField body="enabled" type="boolean">
  是否启用，示例: `True`
</ParamField>


## 响应

<ResponseField name="code" type="integer">
  字段说明
</ResponseField>

<ResponseField name="data" type="object">
  字段说明
</ResponseField>

<ResponseField name="msg" type="string">
  字段说明
</ResponseField>
