获取首页模块的总数、已启用数、已禁用数等统计信息

## 响应

<ResponseField name="code" type="integer">
  字段说明
</ResponseField>

<ResponseField name="data" type="object">
  首页模块统计概览
  
  <Expandable title="展开详情">
    <ResponseField name="total_count" type="integer">
      模块总数（固定为8）
    </ResponseField>
    <ResponseField name="enabled_count" type="integer">
      已启用模块数
    </ResponseField>
    <ResponseField name="disabled_count" type="integer">
      已禁用模块数
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="msg" type="string">
  字段说明
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://dev-admin.beauty-666.com/api/v1/admgmt/homepage/overview" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://dev-admin.beauty-666.com/api/v1/admgmt/homepage/overview', {
```

```python Python
import requests

url = 'https://dev-admin.beauty-666.com/api/v1/admgmt/homepage/overview'
```
